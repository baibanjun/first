变量的解构赋值

//按模式匹配，匹配不上为undefind let [a, b, c] = [1, 2, 3]; let [foo, [[bar], baz]] = [1, [[2], 3]]; let [x, , y] = [1, 2, 3]; //x = 1 , y = 3 let [head, ...tail] = [1, 2, 3, 4]; //head = 1,tail = [2,3,4] let [x, y, ...z] = ['a']; // x = 'a' ,y = undefind ,z = [] let [foo] = []; //undefind let [bar, foo] = [1]; //foo = undefind let [x, y] = [1, 2, 3]; //不完全解构 let [a, [b], d] = [1, [2, 3], 4]; //b = 2 let [foo] = 1; //右边不是数组报错 let [x, y, z] = new Set(['a', 'b', 'c']); //Set结构可以解构 let [x, y = 'b'] = ['a']; // 允许指定默认值 x='a', y='b' let [x, y = 'b'] = ['a', undefined]; // x='a', y='b' 严格匹配（===）undefind时默认值生效 let [x = 1] = [null]; //x = null // //
