cookie及localStorage使用
cookie

//没有设置有效期，则为会话cookie。 $.cookie('cookieName','value',{expires:7,path:'/'}) //读取cookie，存在则返回值，不存在返回null $.cookie('cookieName') //删除cookie，将有效期改为-1 $.cookie('cookieName',null,{expires:-1,path:'/'})

localStorage

//判断是否支持 if(window.localStorage) //写入、读取、移除、清空 localStorage.setItem(key, value); var xx = localStorage.getItem(key); localStorage.removeItem(key); localStorage.clear(); //获取整个localStorage并可按对象处理 var storage = window.localStorage; storage.key1 = "hello"; storage["key2"] = "world"; console.log(storage.key1); console.log(storage["key2"]); //改值或清空会触发storage事件 window.addEventListener("storage",function(){},false);
